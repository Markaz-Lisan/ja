```dataview
TABLE
    المستوى as "المستوى", 
    الفصل as "الفصل", 
    الدرس as "الدرس", 
    نوع_المفردة as "نوع المفردة"
FROM "المنهج/مفردات"
WHERE
	(this.المستوى = null OR المستوى = this.المستوى) AND
	(this.الفصل = null OR الفصل = this.الفصل) AND
	(this.الدرس = null OR الدرس = this.الدرس) AND
	(this.نوع_المفردة = null OR نوع_المفردة = this.نوع_المفردة)
SORT "الدرس" ASC
```
